# Bind TO Firebase
var ExampleComponent = React.createClass({
  mixins: [ReactFireMixin],
  ...
});

# keep in sync with data base using
componentWillMount: function() {
  var ref = new Firebase("https://<YOUR-FIREBASE-APP>.firebaseio.com/items");
  this.bindAsArray(ref, "items");
}
# above: change will be automatically reflected in this.state.items (from "items")
#can use:
bindAsArray() or bindAsObject()

# to unbind early while still in scope:
this.unbind("items") [for a "this.state.items" bind]
